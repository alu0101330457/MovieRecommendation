28      private double getAverageByID(String movieID, int minimalRaters){
29          /* This method returns a double representing the average movie rating for this ID 
30           * if there are at least minimalRaters ratings. 
31             If there are not minimalRaters ratings, then it returns 0.0.*/
32          int numRatings = 0;
33          double totalScore = 0;
34          for (Rater currRater: RaterDatabase.getRaters()){
35              ArrayList<String> currMovies = currRater.getItemsRated();
36              for (String s: currMovies){
37                  if (s.equals(movieID)){
38                      numRatings += 1;
39                      totalScore += currRater.getRating(movieID);
40                  }
41              }
42          }
43          
44          if (numRatings < minimalRaters){
45              return 0.0;
46          } else {
47              return totalScore/numRatings;
48          }
49      }

53           * This method should find the average rating for every movie that has been rated by at least
54           * minimalRaters raters. Store each such rating in a Rating object in which the movie ID 
55           * and the average rating are used in creating the Rating object.

62          // Get the ArrayList of Movies from MovieDatabase.
63          ArrayList<String> movies = MovieDatabase.filterBy(new TrueFilter());

88      private double dotProduct(Rater me, Rater r){
89          /*
90           * This method should first translate a rating from the scale 0 to 10 to the scale -5 to 5 
91           * and return the dot product of the ratings of movies that they both rated. 
92           * This method will be called by getSimilarities.
93           */
94          double result = 0.0;
95          ArrayList<String> myMovieIDs = me.getItemsRated();
96          ArrayList<String> otherMovieIDs = r.getItemsRated();
97          for (String s: myMovieIDs){
98              if (otherMovieIDs.contains(s)){
99                  double myValue = me.getRating(s) - 5;
100                 double otherValue = r.getRating(s) - 5;
101                 result += myValue*otherValue;
102             }
103         }
104         return result;
105     }

107     private ArrayList<Rating> getSimilarities(String id){
108         /*
109          * this method computes a similarity rating for each rater in the RaterDatabase 
110          * (except the rater with the ID given by the parameter) to see how similar they are to the Rater 
111          * whose ID is the parameter to getSimilarities. 
112          * This method returns an ArrayList of type Rating sorted by ratings 
113          * from highest to lowest rating with the highest rating first and only including those raters 
114          * who have a positive similarity rating since those with negative values are not similar in any way. 
115          * Note that in each Rating object the item field is a raterâ€™s ID, 
116          * and the value field is the dot product comparison between that rater and the rater 
117          * whose ID is the parameter to getSimilarities. 
118          * Be sure not to use the dotProduct method with parameter id and itself!
119          */
120         ArrayList<Rating> list = new ArrayList<Rating>();
121         Rater me = RaterDatabase.getRater(id);
122         for (Rater r: RaterDatabase.getRaters()){
123             String currOtherID = r.getID();
124             if (!currOtherID.equals(id)){
125                 double currDotProduct = dotProduct(me, r);
126                 if (currDotProduct > 0.0){
127                     list.add(new Rating(currOtherID, currDotProduct));
128                 }
129             }
130         }
131         Collections.sort(list, Collections.reverseOrder());
132         // The instance variables in every object of the arraylist are ID(String) and dotProduct(double).
133         return list; 
134     }

136     public ArrayList<Rating> getSimilarRatings(String id, int numSimilarRaters, int minimalRaters){
137         /*
138          * This method should return an ArrayList of type Rating, of movies and their weighted average ratings 
139          * using only the top numSimilarRaters with positive ratings and including only those movies 
140          * that have at least minimalRaters ratings from those most similar raters (not just minimalRaters ratings overall). 
141          * For example, if minimalRaters is 3 and a movie has 4 ratings but only 2 of those ratings 
142          * were made by raters in the top numSimilarRaters, that movie should not be included. 
143          * These Rating objects should be returned in sorted order by weighted average rating from largest to smallest ratings. 
144          * This method is very much like the getAverageRatings method you have written previously.
145          */
146         ArrayList<Rating> similarList = getSimilarities(id);
147         // System.out.println("Similar raters list: " + similarList); // Test
148         // Key: Movies' IDs.  Value: RaterID and ratring value.
149         HashMap<String, HashMap<String, Double>> recMap = new HashMap<String, HashMap<String, Double>>();
150         if (similarList.size() < numSimilarRaters){
151             numSimilarRaters = similarList.size();
152         }
153         for (int k=0; k<numSimilarRaters; k++){
154             String currRaterID = similarList.get(k).getItem();
155             Rater currRater = RaterDatabase.getRater(currRaterID);
156             ArrayList<String> ratedMovies = currRater.getItemsRated();
157             for (String currMovie: ratedMovies){
158                 if (!recMap.containsKey(currMovie)){
159                     HashMap<String, Double> first = new HashMap<String, Double>();
160                     first.put(currRaterID, currRater.getRating(currMovie));
161                     recMap.put(currMovie, first);
162                 } else {
163                     recMap.get(currMovie).put(currRaterID, currRater.getRating(currMovie));
164                 }
165             }
166         }
167         // System.out.println(recMap);
168         
169         ArrayList<Rating> result = new ArrayList<Rating>();
170         for (String currMovie : recMap.keySet()){
171             HashMap<String, Double> currValueMap = recMap.get(currMovie);
172             if (currValueMap.size() >= minimalRaters){
173                 double total = 0.0;
174                 for (String currRaterID: currValueMap.keySet()){
175                     double currSimilarRating = 0.0;
176                     // Find similar rating for the currRater.
177                     for (Rating r: similarList){
178                         if (r.getItem().equals(currRaterID)){
179                             currSimilarRating = r.getValue();
180                         }
181                     }
182                     total += currValueMap.get(currRaterID)*currSimilarRating;
183                 }
184                 double weightedAverage = total/currValueMap.size();
185                 result.add(new Rating(currMovie, weightedAverage));
186             }
187         }
188         Collections.sort(result, Collections.reverseOrder());
189         return result;
190     }
