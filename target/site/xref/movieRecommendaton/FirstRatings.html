19      public ArrayList<Movie> loadMovies(String filename){
20          ArrayList<Movie> movies = new ArrayList<Movie>();
21          FileResource fr = new FileResource(filename);
22          CSVParser movieParser = fr.getCSVParser();
23          for (CSVRecord currentRow: movieParser){
24              String currId = currentRow.get("id");
25              String currTitle = currentRow.get("title");
26              String currYear = currentRow.get("year");
27              String currGenres = currentRow.get("genre");
28              String currDirector = currentRow.get("director");
29              String currCountry = currentRow.get("country");
30              String currPoster = currentRow.get("poster");
31              int currMinutes = Integer.parseInt(currentRow.get("minutes"));
32              
33              movies.add(new Movie(currId, currTitle, currYear, currGenres, 
34                                  currDirector, currCountry, currPoster, currMinutes));
35          }
36          return movies;
37      }

39      public ArrayList<Rater> loadRaters(String filename){
40          ArrayList<Rater> raters = new ArrayList<Rater>();
41          FileResource fr = new FileResource("data/" + filename);
42          CSVParser raterParser = fr.getCSVParser();
43          // Initial the String to check if the rater is existed or not. 
44          String existed = "";
45          for (CSVRecord currentRow: raterParser){
46              String raterId = currentRow.get("rater_id");
47              String movieId = currentRow.get("movie_id");
48              double rating = Double.parseDouble(currentRow.get("rating"));
49              
50              // Ckeck if the current rater is existed in the ArrayList "raters" or not.
51              boolean exist = false;
52              /*for (Rater r: raters){
53                  if (r.getID().equals(raterId)){
54                      exist = true;
55                  }
56              }*/
57              if (existed.indexOf(raterId) != -1){
58                  exist = true;
59              }
60              
61              if (exist == false){
62                  //Rater currRater = new PlainRater(raterId);
63                  Rater currRater = new EfficientRater(raterId);
64                  currRater.addRating(movieId, rating);
65                  raters.add(currRater);
66                  // Add the rater's ID to the "existed" String.
67                  existed += raterId + " ";
68              } else {
69                  for (Rater r: raters){
70                      if (r.getID().equals(raterId)){
71                          r.addRating(movieId, rating);
72                      }
73                  }
74              }
75          }
76          return raters;
77      }

104     public void findNumOfRater(String filename, String RaterID){
105         ArrayList<Rater> raters = loadRaters(filename);
106         Rater result = null;
107         for (Rater currRater: raters){
108             if (currRater.getID().equals(RaterID)){
109                 result = currRater;
110             }
111         }
112         int num = result.numRatings();
113         System.out.println("There are " + num + " ratings of " + "ID " + RaterID);
114     }

116     public void findMaxNumOfRatingsByRater(String filename){
117         ArrayList<Rater> raters = loadRaters(filename);
118         int max = 0;
119         for (Rater currRater: raters){
120             if (currRater.numRatings() > max){
121                 max = currRater.numRatings();
122             }
123         }
124         System.out.println("The maximum number of ratings of the rater(s) is " + max + ". Their IDs are:");
125         String s = "";
126         for (Rater currRater: raters){
127             if (currRater.numRatings() == max){

129             }

148         ArrayList<String> movies = new ArrayList<String>();

153                     movies.add(s);
