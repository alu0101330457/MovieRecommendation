43      public void printRecommendationsFor (String webRaterID){
44          MovieDatabase.initialize("ratedmoviesfull.csv");
45          FourthRatings fourth = new FourthRatings();
46          ArrayList<Rating> result = fourth.getSimilarRatings(webRaterID, numSimilarRaters, minimalRaters);
47          int num = result.size();
48          if (num == 0){
49              System.out.println("Recommendation List:");
50              System.out.println("Not Found");
51          } else {
52              if (num > maxRecNum){
53                  num = maxRecNum;
54              }
55              String header = ("<table> <tr> <th>Title</th> <th>Rating Value</th> </tr>");
56              String body = "";
57              for (int k=0; k<num; k++){
58                  Rating currRating = result.get(k);
59                  String currMovieID = currRating.getItem();
60                  // System.out.println(MovieDatabase.getTitle(currMovieID) + " : " + currRating.getValue());
61                  String currMovieTitle = MovieDatabase.getTitle(currMovieID);
62                  double currRatingValue = currRating.getValue();
63                  String currGenre = MovieDatabase.getGenres(currMovieID);
64                  body += printOut(currMovieTitle, currRatingValue, currGenre);
65              }
66              System.out.println(header + body + "</table>");
67          }
68      }
69      
70      private String printOut(String title, double value, String genre){
71          return ("<tr> <td>" + title + "</td> <td>" + Double.toString(value) + "</td> </tr>");
72      }
